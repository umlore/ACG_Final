/ test basic rotation
'cube.obj' 
{ 
	start
	p( 1, 1, 1) 
	r( 0, 0, 0)

	interpolate
	r( 30, 30, 30)
	t(2)
}

/ move in a |-->+ shape
/               |
/               v
/           |<--+
'cube.obj'
{
	start
	p(0,0,0)

	interpolate
	p(1,0,0)
	t(2)

	interpolate
	p(1,1,0)
	t(2)

	interpolate
	p(0,1,0)
	t(2)
}

/ test scale
'cube.obj'
{
	start
	p(0,0,0)
	s(1)

	interpolate
	s(4)
	t(5)
}

/ test orbit
{
	start
	p(0,0,0)
	
	orbit
	o(1,1,1)
	n(1,1,1)
	t(5)
}

/ test everything together
{
	start
	f(0) 
	p( 1, 1, 1) 
	r( 30, 30, 30)
	s(4)
	interpolate
	f(1)
	p( 1, 1, 1) 
	r( 30, 30, 30)
	s(4) // scale
	t(5) // time 
	orbit
	o(1, 1, 1)
	n(1, 1, 1)
	t(5)
	goto 0
}
